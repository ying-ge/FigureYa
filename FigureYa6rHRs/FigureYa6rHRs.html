paper

hrzl_linesnrow(data)+5

Read the documentation carefully, adjust the parameters, and youll be able to draw what you see in the paper

Focus on modifying the third line in the hrzl_lines parameter, draw a black line at the bottom, and the number in as nrow(data)+5

forestplot(labeltext=tabletext, # text in figure
           mean=c(NA,NA,1,data$Point.Estimate,NA),# HR
           lower=c(NA,NA,1,data$Low,NA), #95% lower limit of 95% confidence interval
           upper=c(NA,NA,1,data$High,NA),#95% upper limit of 95% confidence interval
           #title="Hazard Ratio",
           graph.pos=3,# column position of the graph in the table
           graphwidth = unit(.4,"npc"),# width ratio of the graph in the table
           fn.ci_norm="fpDrawDiamondCI",#box box type selection diamond
           col=fpColors(box="steelblue", lines="black", zero = "black"),#box box color
           boxsize=c(NA,NA,NA,data$Percent,NA)/75,#box box size is set according to the sample size
           lwd.ci=2,ci.vertices.height = 0.1,ci.vertices=TRUE,# confidence interval with line width, height, and type
           zero=1,#zero horizontal coordinates of the zero line
           lwd.zero=2,#zero zero line width
           grid = structure(c(data[1,]$Point.Estimate), gp = gpar(col = "black", lty=2,lwd=2)),#() dashed lines (can be more than one) and their horizontal coordinates, colors, line widths
           xticks = c(0.5, 1,1.5, 2),# horizontal coordinate scale can be set as freely
           lwd.xaxis=2,#X X-axis line width
           xlab="     <-Favors A Arm      Favors B Arm->",#X X-axis title
           hrzl_lines=list("3" = gpar(lwd=2, col="black"),# add a black line at the top of the third line, with numbers in quotes indicating the line position
                           #"4" = gpar(lwd=60,lineend="butt", columns=c(1:4), col="#99999922"),# add shadows, not recommended for weak items
                           "30" = gpar(lwd=2, col="black")),#,""nrow(data)+5 add a black line at the bottom of the last line, with the number in quotes as nrow(data)+5
           txt_gp=fpTxtGp(label=gpar(cex=1.25),# various font size settings
                          ticks=gpar(cex=1.25),
                          xlab=gpar(cex = 1.25),
                          title=gpar(cex = 1.25)),
           #is.summary = c(T,rep(F,27)),# first line font type setting
           lineheight = unit(.75,"cm"),# fixed row height
           #align=c("l","c","c"),# the alignment of the text in each column is occasionally used to
           #cex=10,
           colgap = unit(0,"cm"),# column gap
           mar=unit(rep(1.25, times = 4), "cm"),# graphic margin
           new_page = F# new page or not
           )
